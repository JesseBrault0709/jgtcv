% cvSection: a section with a bigBold heading followed by a horizontal rule
\def \cvSection#1{
    \bigskip
	\noindent {\bigBold #1}
	\bigskip
	{\bigBold \hrule}
	\bigskip 
	\par
}

% cv items, with dates and optional descriptors
\def \itemAndDate#1#2{\noindent {\bf #1} \hfill {\bf #2}}

\def \datedItem#1#2{\itemAndDate{#1}{#2} \par \bigskip}
\def \datedItemWithDescription#1#2#3{ \itemAndDate{#1}{#2} \smallskip #3 }

\def \descriptor#1{{\noindent \hang \narrower #1 \par }}


\long\def \experience#1#2#3#4{\datedItemWithDescription{#1}{#2}{\descriptor{#3} \descriptor{#4}}}
